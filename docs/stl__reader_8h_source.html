415  // a coordinate triple with an additional index. The index is required
416  // for RemoveDoubles, so that triangles can be reindexed properly.
417  template <typename number_t, typename index_t>
418  struct CoordWithIndex {
419  number_t data[3];
420  index_t index;
421 
422  bool operator == (const CoordWithIndex& c) const
423  {
424  return (c[0] == data[0]) && (c[1] == data[1]) && (c[2] == data[2]);
425  }
426 
427  bool operator != (const CoordWithIndex& c) const
428  {
429  return (c[0] != data[0]) || (c[1] != data[1]) || (c[2] != data[2]);
430  }
431 
432  bool operator < (const CoordWithIndex& c) const
433  {
434  return (data[0] < c[0])
435  || (data[0] == c[0] && data[1] < c[1])
436  || (data[0] == c[0] && data[1] == c[1] && data[2] < c[2]);
437  }
438 
439  inline number_t& operator [] (const size_t i) {return data[i];}
440  inline number_t operator [] (const size_t i) const {return data[i];}
441  };

443  // sorts the array coordsWithIndexInOut and copies unique indices to coordsOut.
444  // Triangle-corners are re-indexed on the fly and degenerated triangles are removed.
445  template <class TNumberContainer, class TIndexContainer>
446  void RemoveDoubles (TNumberContainer& uniqueCoordsOut,
447  TIndexContainer& trisInOut,
448  std::vector <CoordWithIndex<
449  typename TNumberContainer::value_type,
450  typename TIndexContainer::value_type> >
451  &coordsWithIndexInOut)
452  {
453  using namespace std;
454 
455  typedef typename TNumberContainer::value_type number_t;
456  typedef typename TIndexContainer::value_type index_t;
457 
458  sort (coordsWithIndexInOut.begin(), coordsWithIndexInOut.end());
459 
460  // first count unique indices
461  size_t numUnique = 1;
462  for(size_t i = 1; i < coordsWithIndexInOut.size(); ++i){
463  if(coordsWithIndexInOut[i] != coordsWithIndexInOut[i - 1])
464  ++numUnique;
465  }
466 
467  uniqueCoordsOut.resize (numUnique * 3);
468  vector<index_t> newIndex (coordsWithIndexInOut.size());
469 
470  // copy unique coordinates to 'uniqueCoordsOut' and create an index-map
471  // 'newIndex', which allows to re-index triangles later on.
472  size_t curInd = 0;
473  newIndex[0] = 0;
474  for(size_t i = 0; i < 3; ++i)
475  uniqueCoordsOut[i] = coordsWithIndexInOut[0][i];
476 
477  for(size_t i = 1; i < coordsWithIndexInOut.size(); ++i){
478  const CoordWithIndex <number_t, index_t> c = coordsWithIndexInOut[i];
479  if(c != coordsWithIndexInOut[i - 1]){
480  ++curInd;
481  for(size_t j = 0; j < 3; ++j)
482  uniqueCoordsOut[curInd * 3 + j] = coordsWithIndexInOut[i][j];
483  }
484 
485  newIndex[c.index] = curInd;
486  }
487 
488  // re-index triangles, so that they refer to 'uniqueCoordsOut'
489  // make sure to only add triangles which refer to three different indices
490  size_t numUniqueTriInds = 0;
491  for(size_t i = 0; i < trisInOut.size(); i+=3){
492  int ni[3];
493  for(int j = 0; j < 3; ++j)
494  ni[j] = newIndex[trisInOut[i+j]];
495 
496  if((ni[0] != ni[1]) && (ni[0] != ni[2]) && (ni[1] != ni[2])){
497  for(int j = 0; j < 3; ++j)
498  trisInOut[numUniqueTriInds + j] = ni[j];
499  numUniqueTriInds += 3;
500  }
501  }
502 
503  if(numUniqueTriInds < trisInOut.size())
504  trisInOut.resize (numUniqueTriInds);
505  }
506 }// end of namespace stl_reader_impl

523 template <class TNumberContainer, class TIndexContainer>
524 bool ReadStlFile_ASCII(const char* filename,
525  TNumberContainer& coordsOut,
526  TNumberContainer& normalsOut,
527  TIndexContainer& trisOut,
528  TIndexContainer& solidRangesOut)
529 {
530  using namespace std;
531  using namespace stl_reader_impl;
532 
533  typedef typename TNumberContainer::value_type number_t;
534  typedef typename TIndexContainer::value_type index_t;
535 
536  coordsOut.clear();
537  normalsOut.clear();
538  trisOut.clear();
539  solidRangesOut.clear();
540 
541  ifstream in(filename);
542  STL_READER_COND_THROW(!in, "Couldn't open file " << filename);
543 
544  vector<CoordWithIndex <number_t, index_t> > coordsWithIndex;
545 
546  string buffer;
547  vector<string> tokens;
548  int lineCount = 1;
549  int maxNumTokens = 0;
550  size_t numFaceVrts = 0;
551 
552  while(!(in.eof() || in.fail()))
553  {
554  // read the line and tokenize.
555  // In order to reuse memory in between lines, 'tokens' won't be cleared.
556  // Instead we count the number of tokens using 'tokenCount'.
557  getline(in, buffer);
558 
559  istringstream line(buffer);
560  int tokenCount = 0;
561  while(!(line.eof() || line.fail())){
562  if(tokenCount >= maxNumTokens){
563  maxNumTokens = tokenCount + 1;
564  tokens.resize(maxNumTokens);
565  }
566  line >> tokens[tokenCount];
567  ++tokenCount;
568  }
569 
570  if(tokenCount > 0)
571  {
572  string& tok = tokens[0];
573  if(tok.compare("vertex") == 0){
574  if(tokenCount < 4){
575  STL_READER_THROW("ERROR while reading from " << filename <<
576  ": vertex not specified correctly in line " << lineCount);
577  }
578 
579  // read the position
580  CoordWithIndex <number_t, index_t> c;
581  for(size_t i = 0; i < 3; ++i)
582  c[i] = atof(tokens[i+1].c_str());
583  c.index = static_cast<index_t>(coordsWithIndex.size());
584  coordsWithIndex.push_back(c);
585  ++numFaceVrts;
586  }
587  else if(tok.compare("facet") == 0)
588  {
589  STL_READER_COND_THROW(tokenCount < 5,
590  "ERROR while reading from " << filename <<
591  ": triangle not specified correctly in line " << lineCount);
592 
593  STL_READER_COND_THROW(tokens[1].compare("normal") != 0,
594  "ERROR while reading from " << filename <<
595  ": Missing normal specifier in line " << lineCount);
596 
597  // read the normal
598  for(size_t i = 0; i < 3; ++i)
599  normalsOut.push_back (atof(tokens[i+2].c_str()));
600 
601  numFaceVrts = 0;
602  }
603  else if(tok.compare("outer") == 0){
604  STL_READER_COND_THROW ((tokenCount < 2) || (tokens[1].compare("loop") != 0),
605  "ERROR while reading from " << filename <<
606  ": expecting outer loop in line " << lineCount);
607  }
608  else if(tok.compare("endfacet") == 0){
609  STL_READER_COND_THROW(numFaceVrts != 3,
610  "ERROR while reading from " << filename <<
611  ": bad number of vertices specified for face in line " << lineCount);
612 
613  trisOut.push_back(coordsWithIndex.size() - 3);
614  trisOut.push_back(coordsWithIndex.size() - 2);
615  trisOut.push_back(coordsWithIndex.size() - 1);
616  }
617  else if(tok.compare("solid") == 0){
618  solidRangesOut.push_back(trisOut.size() / 3);
619  }
620  }
621  lineCount++;
622  }
623 
624  solidRangesOut.push_back(trisOut.size() / 3);
625 
626  RemoveDoubles (coordsOut, trisOut, coordsWithIndex);
627 
628  return true;
629 }

632 template <class TNumberContainer, class TIndexContainer>
633 bool ReadStlFile_BINARY(const char* filename,
634  TNumberContainer& coordsOut,
635  TNumberContainer& normalsOut,
636  TIndexContainer& trisOut,
637  TIndexContainer& solidRangesOut)
638 {
639  using namespace std;
640  using namespace stl_reader_impl;
641 
642  typedef typename TNumberContainer::value_type number_t;
643  typedef typename TIndexContainer::value_type index_t;
644 
645  coordsOut.clear();
646  normalsOut.clear();
647  trisOut.clear();
648  solidRangesOut.clear();
649 
650  ifstream in(filename, ios::binary);
651  STL_READER_COND_THROW(!in, "Couldnt open file " << filename);
652 
653  char stl_header[80];
654  in.read(stl_header, 80);
655  STL_READER_COND_THROW(!in, "Error while parsing binary stl header in file " << filename);
656 
657  unsigned int numTris = 0;
658  in.read((char*)&numTris, 4);
659  STL_READER_COND_THROW(!in, "Couldnt determine number of triangles in binary stl file " << filename);
660 
661  vector<CoordWithIndex <number_t, index_t> > coordsWithIndex;
662 
663  for(unsigned int tri = 0; tri < numTris; ++tri){
664  float d[12];
665  in.read((char*)d, 12 * 4);
666  STL_READER_COND_THROW(!in, "Error while parsing trianlge in binary stl file " << filename);
667 
668  for(int i = 0; i < 3; ++i)
669  normalsOut.push_back (d[i]);
670 
671  for(size_t ivrt = 1; ivrt < 4; ++ivrt){
672  CoordWithIndex <number_t, index_t> c;
673  for(size_t i = 0; i < 3; ++i)
674  c[i] = d[ivrt * 3 + i];
675  c.index = static_cast<index_t>(coordsWithIndex.size());
676  coordsWithIndex.push_back(c);
677  }
678 
679  trisOut.push_back(coordsWithIndex.size() - 3);
680  trisOut.push_back(coordsWithIndex.size() - 2);
681  trisOut.push_back(coordsWithIndex.size() - 1);
682 
683  char addData[2];
684  in.read(addData, 2);
685  STL_READER_COND_THROW(!in, "Error while parsing additional triangle data in binary stl file " << filename);
686  }
687 
688  solidRangesOut.push_back(0);
689  solidRangesOut.push_back(trisOut.size() / 3);
690 
691  RemoveDoubles (coordsOut, trisOut, coordsWithIndex);
692 
693  return true;
694 }
